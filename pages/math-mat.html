5.数学-矩阵（附带快速幂）

注意：按需设置数组大小！

class Matrix {
        int dat[101][101];
        int size;
    public:
        Matrix(int size_=0) {
            memset(dat,0,sizeof(dat));
            size=size_;
        }
        void set_init(){
            for(int i=1;i<=size;i++)
                dat[i][i]=1;
        }
        void set(int x,int y,int v) {
            dat[x][y]=v;
        }
        int get(int x,int y) {
            return dat[x][y]%MOD;
        }
        friend Matrix operator * (Matrix &a_,Matrix &b_){
            Matrix ans_(a_.size);
            for(int i=1; i<=a_.size; i++) {
                for(int j=1; j<=a_.size; j++) {
                    for(int k=1; k<=a_.size; k++) {
                        ans_.dat[i][j]+=a_.dat[i][k]*b_.dat[k][j]%MOD;
                        ans_.dat[i][j]%=MOD;
                    }
                }
            }
            return ans_;
        }
        friend Matrix operator ^ (Matrix n_,int k_){
            if(k_==1)return n_;
            Matrix ans_(n_.size);
            ans_.set_init();
            while(k_) {
                if(k_&1)
                    ans_=ans_*n_;
                n_=n_*n_;
                k_>>=1;
            }
            return ans_;
        }
};

乘法时间复杂度：O(n^3)

次方时间复杂度：O(n^3logk)
